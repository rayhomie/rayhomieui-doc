<Props of={Input} />

## 基础用法

<Playground>
  {() => {
    const [value, setValue] = React.useState("");
    return (
      <Input
        placeholder="请输入些什么..."
        value={value}
        onChange={(e) => {
          setValue(e.target.value);
        }}
        style={{ width: "240px" }}
      />
    );
  }}
</Playground>
